# textcursor

```javascript

import {TextCursor} from './textcursor.js';
txtcrsr = new TextCursor("button", "click button for a good time");

txtcrsr.fontFillStyle = "black";
txtcrsr.bgFillStyle = "Yellow";
txtcrsr.bgGlobalAlpha = 0.6;
txtcrsr.borderStrokeStyle = "black";
txtcrsr.face = "sans-serif";
txtcrsr.width = 128;
txtcrsr.height = 32;
txtcrsr.scale = .3;
txtcrsr.pointer_height = 6;

txtcrsr.set_text('hello world');

```
